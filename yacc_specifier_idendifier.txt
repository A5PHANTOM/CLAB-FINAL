YACC
%{
#include<stdio.h>
int valid=1;
%}
%token digit letter
%%
start:letter s
s:letter s
|digit s
|
;
%%
int yyerror(){
printf("\nIt's not an identifier!\n");
valid=0;
return 0;
}
int main(){
printf("\nEnter a name to tested for identifier\n");
yyparse();
if(valid){
printf("\nIt is an identifier!\n");
}
}

%{
#include"y.tab.h"
%}
%%
[a-zA-Z_][a-zA-Z_0-9]* return letter; /*identifiers*/

. return yytext[0]; /*return character itself*/
\n return 0; /*end of input*/
%%
int yywrap(){
return 1;
}
